Bresenham 画线算法

def draw_line(    v1: Vec2d, v2: Vec2d, canvas: Canvas):    dots = []    err = 0    y = v1.y    slope = abs((v1.y - v2.y) / (v1.x - v2.x))    for x in range(v1.x, v2.x):        dots.append(x, y)        err += slope        if abs(err) >= 0.5:            y += 1            err -= 1     canvas.draw(dots)
